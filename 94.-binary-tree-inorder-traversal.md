var traverseInOrder = function (node,list){
    if (node.left) {
        traverseInOrder(node.left, list);
    }
    list.push(node.val);
    if (node.right) {
        traverseInOrder(node.right, list);
    }
    return list;
}

```javascript
var inorderTraversal = function(root) {
    if (root === null) return [];
    const stack = [];
    const values = [];
    let curr = root;
    
    while (curr !== null || stack.length !== 0) {
        while (curr !== null) {
            stack.push(curr);
            curr = curr.left;
        } 
        curr = stack.pop();
        values.push(curr.val);
        
        curr = curr.right;
    }
    return values;
};
```
