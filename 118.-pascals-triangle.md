## My Code

```javascript
/**
 * @param {number} numRows
 * @return {number[][]}
 */
var generate = function(numRows) {
    let pas = [];
    let Row = [1];
    if(numRows>=1){
        pas.push(Row)
    }
    for(let i = 1; i < numRows; i++){
        let nextRow = nextLevel(Row);
        pas.push(nextRow);
        Row = nextRow;  
    }
    
    return pas
};

var nextLevel = function(Row){
    let nextRow = [1];
    for(let i = 0; i < Row.length-1; i++){
        nextRow.push(Row[i]+Row[i+1])
    }
    nextRow.push(1);
    return nextRow;
}
```
