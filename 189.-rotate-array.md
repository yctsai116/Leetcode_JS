# 189. Rotate Array

![](.gitbook/assets/image%20%2853%29.png)

## My Code

```javascript
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var rotate = function(nums, k) {
    while(k){
        nums.unshift(nums.slice(-1));
        nums.pop()
        k--;
    }
};

var rotate = function(nums, k) {
    nums.unshift(...nums.splice(-(k%nums.length)));
};
```
