## My Code

```javascript
/**
 * @param {number} n
 * @return {boolean}
 */
 
 // Do it w/o loop/recursion ( result is quite slow)
var isPowerOfThree = function(n) {
    if(n===0){
        return false
    };
    
    var arr = n.toString(3).split("");
    
    if(arr[0]!=="1"){
        return false
    }

    return arr.indexOf("1",1)===-1 && arr.indexOf("2",1)===-1
};

// w/ loop

var isPowerOfThree = function(n) {
    if(n===1){
        return true
    };
    while(n>3){
        n=n/3
    }
    return n===3
};
```
