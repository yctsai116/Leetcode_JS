# 283. Move Zeroes

![](.gitbook/assets/image%20%281%29.png)

## My Code

```javascript
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var moveZeroes = function(nums) {
    // how many non-zeros
    let n = 0;
    
    for(let i = 0; i < nums.length; i++){
        if(nums[i]!==0){
            nums[n]=nums[i];
            n++;
        }
    }
    nums.fill(0, n)
    return nums
};
```
